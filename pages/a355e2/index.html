# 1. shutil copyfile() 方法

copyfile(source_file, destination_file)
1
2
  • 将source_file的内容复制到destination_file
  • 如果目标文件destination_file不可写，将会返回IOErro异常。
  • 如果源文件source_file和目标文件destina_file同名则返回SameFileErro异常
  • 如果目标文件是一个目录则返回Error 13
  • 不支持复制字符、块设备或者管道文件
# Python Copy File - Sample Code
        
from shutil import copyfile
from sys import exit

source = input("Enter source file with full path: ")
target = input("Enter target file with full path: ")

# adding exception handling
try:
    copyfile(source, target)
except IOError as e:
    print("Unable to copy file. %s" % e)
    exit(1)
except:
    print("Unexpected error:", sys.exc_info())
    exit(1)

print("\nFile copy done!\n")

while True:
    print("Do you like to print the file ? (y/n): ")
    check = input()
    if check == 'n':
        break
    elif check == 'y':
        file = open(target, "r")
        print("\nHere follows the file content:\n")
        print(file.read())
        file.close()
        print()
        break
    else:
        continue

# 2. shutil copy() 方法

copyfile(source_file, [destination_file or dest_dir])
1

copy()方法类似于linux系统里的cp命令，如果目标文件名是一个目录，就会在该目录下创建一个相同的名字的文件。

import os
import shutil

source = 'current/test/test.py'
target = '/prod/new'

assert not os.path.isabs(source)
target = os.path.join(target, os.path.dirname(source))

# create the folders if not already exists
os.makedirs(target)

# adding exception handling
try:
    shutil.copy(source, target)
except IOError as e:
    print("Unable to copy file. %s" % e)
except:
    print("Unexpected error:", sys.exc_info())

copy()与copyfile()对比

  • copy()在复制的时候会把源文件的权限一起复制，copyfile()只复制内容
  • copy()在目标文件为目录的时候仍会复制，会在目录下创建个同名的文件。copyfile()会报Error13错误
  • copyfile()方法是用copyfileobj()方法实现的，copy方法是用copyfile()和copymode()函数实现的。

# 3. shutil copyfileobj() method

from shutil import copyfileobj
status = False
if isinstance(target, string_types):
    target = open(target, 'wb')
    status = True
try:
    copyfileobj(self.stream, target, buffer_size)
finally:
    if status:
        target.close()

# 4. shutil copy2() method

copy2()方法与copy()方法类似。但是它在复制文件的时候会把访问和修改时间这种元数据也同时复制到目标文件。

from shutil import *
import os 
import time
from os.path import basename

def displayFileStats(filename):
    file_stats = os.stat(basename(filename))
    print('\tMode    :', file_stats.st_mode)
    print('\tCreated :', time.ctime(file_stats.st_ctime))
    print('\tAccessed:', time.ctime(file_stats.st_atime))
    print('\tModified:', time.ctime(file_stats.st_mtime))

os.mkdir('test')

print('SOURCE:')
displayFileStats(__file__)

copy2(__file__, 'testfile')

print('TARGET:')
displayFileStats(os.path.realpath(os.getcwd() + './test/testfile'))

copymode() vs copystat() copymode()不会复制访问时间、使用时间、更新时间等信息

copymode(source, target, *, follow_symlinks=True)
1
2

copystat()会复制访问时间、使用时间、更新时间等信息

copystat(source, target, *, follow_symlinks=True)
1
2

如果follow_symlinks参数为false并且前面2个参数都是链接，则会只更新链接的信息不更新他们具体指向的文件。

# 8.用子进程的call() 方法复制文件

子进程模块提供留简单的接口来使用子进程。可以让我们启动子进程、附加输入输出或者错误信息 、管道文件，检索返回值。 留出了call()方法调用系统命令来执行用户任务

import subprocess

src="1.txt.py"
dst="2.txt.py"
cmd='copy "%s" "%s"' % (src, dst)

status = subprocess.call(cmd, shell=True)

 if status != 0:
     if status < 0:
         print("Killed by signal", status)
     else:
         print("Command failed with return code - ", status)
else:
     print('Execution of %s passed!\n' % cmd)

# 9.使用子进程的check_output()方法复制文件

用subprocess’s check_output() 方法，可以执行外外部的命令或者程序并且补货它的输出，同时也直至管道pipes。

import os, subprocess

src=os.path.realpath(os.getcwd() + "./1.txt.py")
dst=os.path.realpath(os.getcwd() + "./2.txt.py")
cmd='copy "%s" "%s"' % (src, dst)

status = subprocess.check_output(['copy', src, dst], shell=True)

print("status: ", status.decode('utf-8'))
