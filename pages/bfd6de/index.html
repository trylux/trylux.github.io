    # 3. LinkedList.c 线性表的单链表实现

    
/*-------------------------------------
 *
 *
 *
 *   线性表(单链表)
 *
 *
 *
 *
 *--------------------------------------*/

#ifndef SINGLELINKEDLIST_C
#define SINGLELINKEDLIST_C 

#include "../Include/SingleLinkedList.h"


//初始化单链表
Status InitList_Lk(LinkedList *L)
{
    (*L)=(LinkedList)malloc(sizeof(LNode));
    if(!(*L))           //检查内存是否分配成功
        exit(OVERFLOW);
    (*L)->next=NULL;

    return OK;
}

//置空单链表,头结点保留
Status ClearList_Lk(LinkedList L)
{
    LinkedList pre,p;

    if(!L)              //链表为空，返回错误
        return ERROR;

    pre=L->next;        //pre指向L第一个结点

    while(pre)          //顺指针向后逐个释放结点
    {
        p=pre->next;
        free(pre);      //释放结点
        pre=p;
    }

    L->next=NULL;

    return OK;
}

//销毁单链表,连同头结点一起销毁
void DestroyList_Lk(LinkedList *L)
{
    LinkedList p=*L;

    while(p)
    {
        p=(*L)->next;
        free(*L);
        (*L)=p;
    }
}

//判断单链表L是否为空
Status ListEmpty_Lk(LinkedList L)
{
    if(L!=NULL && L->next==NULL)
        return TRUE;
    else 
        return FALSE;
}

//返回单链表L的元素个数
int ListLength_Lk(LinkedList L)
{
    int i=0;
    LNode *p;

    if(L)
    {
        p=L->next;

        while(p)
        {
            i++;
            p=p->next;
        }
    }

    return i;
}
/*---------------算法2.8---------------*/
//用e接收单链表L中第i个元素
Status GetElem_Lk(LinkedList L,int i,ElemType *e)
{
    int j=1;
    LNode *p=L->next;   //初始化,p指向第一个结点,j为计数器

    while(p && j<i)     //顺指针向后查找，直到p指向第i个元素或p为空
    {
        j++;
        p=p->next;
    }

    if(!p || j>i)       //第i个元素不存在
        return ERROR;

    *e=p->data;         //取第i个元素

    return OK;
}

//返回单链表中第一个与e满足Compare关系的元素位序
int LocateElem_Lk(LinkedList L,ElemType e,Status(Comapre)(ElemType,ElemType))
{
    int i;
    LNode *p;

    i=-1;           //L不存在时返回-1；
    if(L)
    {
        i=0;
        p=L->next;

        while(p)
        {
            i++;

            if(!Comapre(e,p->data))
            {
                p=p->next;
                if(p==NULL)
                    i=0;       //没有匹配的元素时返回0
            }
            else 
                break;
        }
    }

    return i;
}

//用pre_e接收cur_e的前驱
Status PriorElem_Lk(LinkedList L,ElemType cur_e,ElemType *pre_e)
{
    LNode *p,*suc;

    if(L)
    {
        p=L->next;

        if(p->data!=cur_e)          //第一个结点无前驱
        {
            while(p->next)
            {
                suc=p->next;
                if(suc->data==cur_e)
                {
                    *pre_e=p->data;
                    return OK;
                }
                p=suc;
            }
        }
    }

    return ERROR;
}

//用next_e接收cur_e的后继
Status NextElem_Lk(LinkedList L,ElemType cur_e,ElemType *next_e)
{
    LNode *p,*suc;

    if(L)
    {
        p=L->next;             

        while(p && p->next)             //当L有且不只一个结点时
        {
            suc=p->next;

            if(p->data==cur_e)
            {
                *next_e=suc->data;
                return OK;
            }
            p=suc;
        }
    }

    return ERROR;
}

/*------------算法2.9-----------------------*/
//在单链表L第i个位置之前插入e
Status ListInsert_Lk(LinkedList L,int i,ElemType e)
{
    LNode *p,*s;
    int j=0;
    p=L;

    while(p && j<i-1)           //寻找第i-1个结点
    {
        p=p->next;
        ++j;
    }

    if(!p || j>i-1)
        return ERROR;

    s=(LinkedList)malloc(sizeof(LNode));        //生成新结点

    //插入L中
    s->data=e;  
    s->next=p->next;
    p->next=s;

    return OK;
}

/*------------算法2.10----------------------*/
//删除单链表L第i个位置的元素，并用e接收其值
Status ListDelete_Lk(LinkedList L,int i,ElemType *e)
{
    LNode *p,*q;
    int j;

    p=L;
    j=1;

    while(p->next && j<i)     //寻找第i个结点，并令pre指向其前驱
    {
        p =p->next;
        ++j;
    }

    if(!p->next || j>i)        //删除位置不合理
        return ERROR;

    //删除并释放结点
    q=p->next;
    p->next=q->next;
    *e=q->data;
    free(q);


    return OK;

}

//用Visit函数访问单链表L
Status ListTraverse_Lk(LinkedList L,void(Visit)(ElemType e))
{
    LNode *p;

    if(!L)                      //如果L不存在
        return ERROR;
    else 
        p=L->next;              //p指向第一个元素

    while(p)
    {
        Visit(p->data);
        p=p->next;
    }

    return OK;
}

/*-----------------算法2.11--------------------------*/
//头插法建立单链表L(逆序插入)
Status CreateList_HL(LinkedList *L,int n)
{
    int i;
    LNode *p;
    *L=(LinkedList)malloc(sizeof(LNode));
    if(!(*L))
        exit(OVERFLOW);
    (*L)->next=NULL;                        //先建立一个带头结点的单链表

    for(i=0;i<n;++i)
    {
        p=(LinkedList)malloc(sizeof(LNode));//生成新结点
        scanf("%d",&p->data);

        p->next=(*L)->next; 
        (*L)->next=p;
    }

   return OK;
}

//尾插法建立链表L(顺序插入)
Status CreateList_TL(LinkedList *L,int n)
{
    int i;
    LNode *p,*q;

    *L=(LinkedList)malloc(sizeof(LNode));
    if(!(*L))
        exit(OVERFLOW);
    (*L)->next=NULL;

    for(i=0,q=*L;i<n;++i)
    {
        p=(LinkedList)malloc(sizeof(LNode));
        if(!p)
            exit(OVERFLOW);
        scanf("%d",&p->data);

        q->next=p;
        q=q->next;
        
    }
    
    q->next=NULL;               //最后一个结点的next指针为空

    return OK;
}


//将单链表La和Lb合并得到新的单链表Lc,(La和Lb的元素按值非递减排列,合并后的Lc的元素也按值非递减排列
void MergeList_Lk(LinkedList *La, LinkedList *Lb,LinkedList *Lc)
{
    LinkedList pa,pb,pc;

    pa=(*La)->next;
    pb=(*Lb)->next;
    pc=*Lc=*La;                         //用La的头结点作为Lc的头结点

    while(pa && pb)
    {
        if(pa->data<=pb->data)
        {
            pc->next=pa;
            pc=pa;
            pa=pa->next;
        }
        else 
        {
            pc->next=pb;
            pc=pb;
            pb=pb->next;
        }
    }

    pc->next=pa?pa:pb;

    free(*Lb);

    *Lb=NULL;
}
